function onKeyPress(event) {
    if (event.key === 'q' || event.key === 'Q') {
        toggleQROverlay();
        event.preventDefault();
        return;
    }

    if ((event.key === 'd' || event.key === 'D') && isQuestionSlide) {
        toggleVoteDetails();
        event.preventDefault();
    }
}

function highlightCorrectAnswer(slide) {
    const correctAnswer = slide.getAttribute('data-correct-answer');
    const correctRow = voteDisplay.querySelector(
        `[data-option="${correctAnswer}"]`
    );

    if (correctRow) {
        correctRow.classList.add('correct-answer');
    }
}
